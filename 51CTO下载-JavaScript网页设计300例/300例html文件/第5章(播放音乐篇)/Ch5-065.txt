<script language=JavaScript>

IndexLayer = (document.layers) ? 1:0
IndexAll = (document.all) ? 1:0

function PlayMusic() {
  if (IndexLayer)   document.Music.play(true)
  if (IndexAll) {	document.Music.stop();document.Music.play()	}
}

function StopMusic() {
   document.Music.stop()
}

</script>
