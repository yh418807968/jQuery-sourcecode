    for (I=0;I<Message.length;I++){
        MFirst1=MFirst+0.5*I;
        Cosine=Math.cos(MFirst1);
        RM=document.all("L"+I);
        RM.style.posLeft=260+100*Math.sin(MFirst1)+50;
        RM.style.zIndeI=20*Cosine;
        RM.style.fontSize=40+25*Cosine;
        RM.style.color="rgb("+ (200+Cosine*80+50) + ","+ (120+Cosine*80+50) + ",0)";
    }
}
Time=window.setInterval("RoundMessage()",100);
</script>
