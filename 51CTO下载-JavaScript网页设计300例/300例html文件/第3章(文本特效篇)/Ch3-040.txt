<script language=javascript>

layerRef="document.all";
styleRef=".style";
     
function UpDownMessage(which, StartPoint, finishPoint, speed){  
   if (StartPoint < finishPoint){ 
      eval(layerRef + '["' + which +'"]' + styleRef + '.top = StartPoint');
      eval(layerRef + '["' + which +'"]' + styleRef + '.top = StartPoint');
      StartPoint=StartPoint+2; 
      setTimeout("UpDownMessage('"+which+"',"+StartPoint+","+finishPoint+","+speed+")",speed);
    }       
}

</script>

<!-- 案例代码1结束 -->
