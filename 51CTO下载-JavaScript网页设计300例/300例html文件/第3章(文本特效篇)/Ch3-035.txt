function StarFirstArray() {
this.length = StarFirstArray.arguments.length;

this[i] = StarFirstArray.arguments[i];   }
}

<!-- [Step1]: 在此可以改变文本的内容 -->
var ctext = "文本特效篇--闪动的文本";
var x = 0;
<!-- [Step2]: 这里能够变换文本显示的颜色 -->
var color = new StarFirstArray("#FF0000", "#FFCC33", "#0000FF","#33CC00","#CC6666","#6699FF");

document.write('<div id="c"><center>'+ctext+'</center></div>');

function Changecolor(){ 
document.all.c.style.color = color[x];
(x < color.length-1) ? x++ : x = 0;  }

setInterval("Changecolor()",100);
