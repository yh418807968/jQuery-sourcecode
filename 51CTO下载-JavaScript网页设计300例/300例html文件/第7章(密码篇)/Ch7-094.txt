<script language=JavaScript>

function EnterPage() {
   var password;
   var location;
   var root;
   password=this.document.testform.inputbox.value
<!-- [Step1]: 这里可以设置相应的页面控制 -->
   location="http://www."+password+".com.cn"  
   root=window.open('location', 'pass', 'toolbar=yes, location=yes, status=yes, menubar=yes, scrollbars=yes, resizable=yes, copyhistory=no')
   root.location.href = location;
 }

</script>
