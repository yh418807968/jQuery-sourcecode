<script language="JavaScript">

function ViewImage(form) {
ix = form.width.value;
iy = form.height.value;
ifile = form.url.value;
NS = (document.layers) ? 1 : 0;
localurl = ifile.indexOf('http://') == -1;
if (NS && localurl) ifile = "file:///" + ifile;
var win = window.open("","","width="+ix+",height="+iy+",menubar=no,toolbar=no,top=100,left=100");
if (NS) {var sWidth = win.innerWidth;var sHeight = win.innerHeight;} 
else {var sWidth = win.document.body.clientWidth;var sHeight = win.document.body.clientHeight;}

win.document.open();
win.document.write("<html><head></head><body onBlur=this.close();>");
win.document.write('<div style="position:absolute;width:'+ix+'px;height:'+iy+'px;left:0px;top:0px">');
win.document.write("<img src="+ifile+" width=" + ix + " height=" + iy + "></div></body></html>");
win.document.close();
}

</script>
