<script language=JavaScript>

var counter = 0;
function SlidePicture()
{
if (document.all) { picture.style.clip = "rect(0," + counter + ",80,0)"; }
if (document.layers)
{ document.picture.clip.top = 0;
  document.picture.clip.right = counter;
  document.picture.clip.bottom = 80;
  document.picture.clip.left = 0; }
counter += 2;
time_out = setTimeout("SlidePicture()", 50);
if (counter >400){ clearTimeout(time_out); }
}

</script>

<!-- 案例代码1结束 -->
