    document.write(numberstring.substring(0+(point-1),point))}
  else if(point < 20 ){
    document.write("十"+numberstring.substring(0+(point-11),(point-10)))}
  else if(point < 30 ){
    document.write("二十"+numberstring.substring(0+(point-21),(point-20)))}
  else{
    document.write("三十"+numberstring.substring(0+(point-31),(point-30)))}
  }
var today = new Date()
var month = today.getMonth()+1
var date = today.getDate()
var day = today.getDay()

CurrentDate(month)
