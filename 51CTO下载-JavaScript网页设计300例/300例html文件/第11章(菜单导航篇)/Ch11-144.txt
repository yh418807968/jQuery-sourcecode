<script language=JavaScript>

function LinkURL(){
<!-- [Step1]: 这里可以更改链接的网址 -->
  if (document.addressform.addressselect.options[0].selected){window.location.href = "http://www.263.net.cn";} 
   else if (document.addressform.addressselect.options[1].selected) { window.location.href = "http://www.sohu.com";}
     else if (document.addressform.addressselect.options[2].selected) { window.location.href = "http://www.sina.com.cn";} 
        else if (document.addressform.addressselect.options[3].selected) { window.location.href = "http://cn.yahoo.com";}
  return true; 
}

function textValue(){ 
   var addressInteger, addressString 
   addressInteger=document.addressform.addressselect.selectedIndex 
   addressString=document.addressform.addressselect.options[addressInteger].text 
<!-- [Step2]: 在此能够修改文本框中的信息 -->
   document.addressform.addresstext.value = "链接－－ " + addressString
}

</script>

<!-- 案例代码1结束 -->



<!-- 案例代码2开始 -->

<form name="addressform">
<!-- [Step3]: 这里可以设置选择框的行数 -->
<select name="addressselect" onChange="textValue()" multiple size="5">
<!-- [Step4]: 在此能够修改网址名称 -->
   <option>263 在线
   <option>搜 狐 网
   <option>新 浪 网
   <option>雅 虎 网
</select><br>
<!-- [Step5]: 这里可以设置文本框的列长度 -->
<input type="text" name="addresstext" value="" size="20" >
<input type="button" name="addressbutton" value="GO" onClick="LinkURL()">
</form>
