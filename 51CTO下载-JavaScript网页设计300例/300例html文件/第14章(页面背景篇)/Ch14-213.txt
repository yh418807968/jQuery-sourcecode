sunset = new Array("skyblue", "slateblue", "violet", "darkorange", "red", "darkred");

function BGColor(color){
    scheme = color;
    document.bgColor = scheme.c1;
    setTimeout('document.bgColor = scheme.c2', document.selector.interval.value * 20);
    setTimeout('document.bgColor = scheme.c3', document.selector.interval.value * 30);
    setTimeout('document.bgColor = scheme.c4', document.selector.interval.value * 40);
    setTimeout('document.bgColor = scheme.c5', document.selector.interval.value * 50);
    setTimeout('document.bgColor = scheme.c6', document.selector.interval.value * 60);
}

function Array(c1, c2, c3, c4, c5, c6){
    this.c1 = c1;  this.c2 = c2;  this.c3 = c3;
    this.c4 = c4;  this.c5 = c5;  this.c6 = c6;
}

</script>
