<script language=JavaScript>

function MouseBGColor(){
    document.bgColor = getColor()
}

function getColor(){
    currentdate = new Date()
    bkc = currentdate.getSeconds()
    if (bkc > 44)   bkc = bkc - 45
    else if (bkc > 29)  bkc = bkc - 30
         else if (bkc > 14)  bkc = bkc - 15
<!-- [Step1]: 这里可以设置链接的背景颜色 -->
    if (bkc == 0 )     return "fuschia";
     else if (bkc == 1 )  return "teal";
      else if (bkc == 2 )  return "yellow";
       else if (bkc == 3 )  return "blue";
        else if (bkc == 4 )  return "white";
         else if (bkc == 5 )  return "silver";
          else if (bkc == 6 )  return "lime";
           else if (bkc == 7 )  return "olive";
            else if (bkc == 8 )  return "green";
             else if (bkc == 9 )  return "purple";
              else if (bkc == 10 ) return "gray";
               else if (bkc == 11 ) return "red";
                else if (bkc == 12 ) return "aqua";
                 else if (bkc == 13 ) return "black";
                  else if (bkc == 14 ) return "maroon";
                    else                return "navy";
}

</script>
