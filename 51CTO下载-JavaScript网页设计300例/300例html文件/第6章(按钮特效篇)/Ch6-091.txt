<script language=JavaScript>

var max=0;

function ButtonList(){
  max=ButtonList.arguments.length;
  for (i=0; i<max; i++)  this[i]=ButtonList.arguments[i];
}

<!-- [Step1]: 在此可以设置按钮显示的滚动信息 -->
tl=new ButtonList( "263 在线", "中国人的网上家园");
var x=0; pos=0;
var l=tl[0].length;

function NewTypeButton() {        
  document.form1.typebutton.value=tl[x].substring(0,pos)+"";
  if(pos++==l) { pos=0; setTimeout("NewTypeButton()",1500); x++;  if(x==max) x=0;l=tl[x].length;} 
<!-- [Step2]: 这里能够设置滚动信息的速度，数值大速度慢 -->
  else setTimeout("NewTypeButton()",100);
}

</script>
