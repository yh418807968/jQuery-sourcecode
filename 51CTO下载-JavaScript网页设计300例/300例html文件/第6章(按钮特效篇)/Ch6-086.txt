function ValidationEmail(){
var str=document.validation.emailAddress.value
var filter=/^.+@.+\..{2,3}$/
if (filter.test(str)) index=true
else{ 
<!-- [Step1]: 这里可以设置弹出窗口的警告信息 -->
   alert("输入的 Email 地址不正确！")
   index=false  }
return (index)
}
