<script language=JavaScript>

function WindowShake() {
  for (i = 20; i > 0; i--) {
    for (j = i; j > 0; j--) {
        self.moveBy(i,0);
		self.moveBy(0,i);
        self.moveBy(-i,0); 
        self.moveBy(0,-i);   }   } 
}

</script>

<!-- 案例代码1结束 -->


<!-- 案例代码2开始 -->

<form>
<!-- [Step1]: 这里可以更改按钮的名称 -->
  <input type=button value="振 动" onClick="WindowShake()">
</form>
