<script language=Javascript>

var index=0
<!-- [Step1]: 这里可以修改状态栏显示的滚动信息 -->
var StatuMessage = "状态栏特效篇--滚动特效！"

function StatuScrollMessage() {
   window.status = StatuMessage.substring(index,StatuMessage.length) + StatuMessage.substring(0,index)
   if (index < StatuMessage.length) { index++ }
   else { index = 0 }
<!-- [Step2]: 在此能够设置滚动的速度 -->
   setTimeout("StatuScrollMessage()",100)
}

</script>
